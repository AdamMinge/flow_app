# ----------------------------------------------------------------------- #
# ---------------------------- Set CMake Version ------------------------ #
# ----------------------------------------------------------------------- #
cmake_minimum_required(VERSION 3.11)
# ----------------------------------------------------------------------- #
# ---------------------------- Set files variable ------------------------#
# ----------------------------------------------------------------------- #
set(source_root ${FLOW_SOURCE_DIR}/src/flow/utils/view)
set(include_root ${FLOW_SOURCE_DIR}/include/flow/utils/view)

set(sources)

set(includes ${include_root}/export.h ${include_root}/unselectable_view.h
             ${include_root}/mouse_over_selectable_view.h)
# ----------------------------------------------------------------------- #
# ------------------------- Qt libraries configuration -------------------#
# ----------------------------------------------------------------------- #
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)
# ----------------------------------------------------------------------- #
# -------------------------- Find external libraries ---------------------#
# ----------------------------------------------------------------------- #
find_package(Qt6 REQUIRED COMPONENTS Widgets)
# ----------------------------------------------------------------------- #
# -------------------------- Create utils shared lib ---------------------#
# ----------------------------------------------------------------------- #
flow_add_utils(view SOURCES ${sources} ${includes} DEPENDS_PRIVATE Qt6::Widgets)
# ----------------------------------------------------------------------- #
